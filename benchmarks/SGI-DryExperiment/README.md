# SGI-DryExperiment

This directory contains the ResearchHarness benchmark role overlay for
[SGI-DryExperiment](https://huggingface.co/datasets/InternScience/SGI-DryExperiment).

SGI-DryExperiment is a strict code-completion benchmark. It is not a generic
QA/VQA task, so the generic QA wrappers are not recommended.

python3 run_server.py \

import sys

import numpy as np
from pathlib import Path

# --- Physical Constants ---
G = 6.67430e-11  # Gravitational constant (m^3 kg^-1 s^-2)
C = 2.99792458e8  # Speed of light (m/s)
M_SOLAR = 1.98847e30  # Solar mass (kg)
MPC_TO_M = 3.086e22  # Megaparsec to meter conversion


def generate_chirp_waveform(times: np.ndarray, m1_solar: float, m2_solar: float, dist_mpc: float) -> np.ndarray:
    """
    Generates a simplified, calibrated gravitational wave chirp waveform.
    Tag: [Simulation]

    Args:
        times (np.ndarray): Time array for the waveform.
        m1_solar (float): Mass of the first black hole in solar masses.
        m2_solar (float): Mass of the second black hole in solar masses.
        dist_mpc (float): Luminosity distance to the source in megaparsecs.

    Returns:
        np.ndarray: The generated strain waveform, scaled to a peak of 1.0e-21.

    Examples:
        >>> fs = 4096
        >>> duration = 2
        >>> ts = np.arange(0, duration, 1/fs)
        >>> waveform = generate_chirp_waveform(ts, 36.0, 29.0, 410.0)
        >>> print(f"{np.max(np.abs(waveform)):.1e}")
        1.0e-21
    """
    m1 = m1_solar * M_SOLAR
    m2 = m2_solar * M_SOLAR
    chirp_mass = ((m1 * m2) ** (3 / 5)) / ((m1 + m2) ** (1 / 5))
    dist_m = dist_mpc * MPC_TO_M

    t_c = times[-1]
    time_to_coalescence = t_c - times
    epsilon = 1e-8
    time_to_coalescence = np.maximum(time_to_coalescence, epsilon)

    k = (5 * G * chirp_mass / C ** 3) ** (-5 / 8)
    phi = -2 * (time_to_coalescence ** (5 / 8)) * k
    freq = (k / (8 * np.pi)) * (time_to_coalescence ** (-3 / 8))
    amplitude = (G * chirp_mass / (dist_m * C ** 2)) * (2 * np.pi * G * chirp_mass * freq / C ** 3) ** (2 / 3)

    freq_mask = (freq >= 35) & (freq <= 350)
    unscaled_waveform = amplitude * np.cos(phi)
    unscaled_waveform[~freq_mask] = 0.0

    peak_strain_observed = 1.0e-21
    current_peak = np.max(np.abs(unscaled_waveform))

    if current_peak < epsilon:
        return unscaled_waveform

    scaling_factor = peak_strain_observed / current_peak
    scaled_waveform = unscaled_waveform * scaling_factor

    return scaled_waveform


def generate_gw_data(duration: float, fs: int, m1_solar: float, m2_solar: float, dist_mpc: float, noise_std: float,
                     output_path: str) -> None:
    """
    Generates synthetic gravitational wave data (signal + noise) and saves it.
    Tag: [Data saving]

    Args:
        duration (float): Duration of the data in seconds.
        fs (int): Sampling frequency in Hz.
        m1_solar (float): Mass of the first black hole in solar masses.
        m2_solar (float): Mass of the second black hole in solar masses.
        dist_mpc (float): Luminosity distance in megaparsecs.
        noise_std (float): Standard deviation of the Gaussian noise.
        output_path (str): Path to save the output .npz file.

    Returns:
        None

    Examples:
        >>> generate_gw_data(2.0, 4096, 36.0, 29.0, 410.0, 0.48e-22, "data/gw_data.npz")
    """
    np.random.seed(0+42)
    times = np.arange(0, duration, 1 / fs)
    signal_template = generate_chirp_waveform(times, m1_solar, m2_solar, dist_mpc)

    full_noise = np.random.normal(0, noise_std, int(duration * fs * 1.5))

    signal_end_index = int(duration * fs * 0.7)
    signal_start_index = signal_end_index - len(times)

    if signal_start_index < 0:
        signal_start_index = 0

    end_slice = signal_start_index + len(signal_template)
    full_noise[signal_start_index:end_slice] += signal_template
    final_strain = full_noise[:len(times)]

    Path(output_path).parent.mkdir(parents=True, exist_ok=True)
    np.savez(output_path,
             times=times,
             strain=final_strain,
             fs=fs,
             m1_solar=m1_solar,
             m2_solar=m2_solar,
             dist_mpc=dist_mpc,
             noise_std=noise_std)
    print(f"Data has been generated and saved to: {output_path}")

def load_gw_data(file_path: str) -> dict:
    """
    Loads gravitational wave data from a .npz file.
    Tag: [Data loading]

    Args:
        file_path (str): The path to the .npz data file.

def estimate_psd(noise_strain: np.ndarray, fs: int) -> tuple[np.ndarray, np.ndarray]:
    """
    Estimates the Power Spectral Density (PSD) from a noise time series.
    Tag: [Statistical analysis]

    Args:
        noise_strain (np.ndarray): A time series of pure noise.
        fs (int): The sampling frequency in Hz.

    Returns:
        tuple[np.ndarray, np.ndarray]: A tuple containing frequency array and PSD array.

    Examples:
        >>> noise = np.random.randn(16 * 4096)
        >>> freqs, psd = estimate_psd(noise, 4096)
        >>> print(freqs.shape, psd.shape)
        (4097,) (4097,)
    """
    nperseg = fs * 2
    freqs, psd = welch(noise_strain, fs=fs, nperseg=nperseg, window='hann')
    return freqs, psd


def matched_filter(data: np.ndarray, template: np.ndarray, fs: int, freqs: np.ndarray, psd: np.ndarray) -> np.ndarray:
    """
    Performs matched filtering of data with a template in the frequency domain.
    Tag: [Filtering]

    Args:
        data (np.ndarray): The data time series.
        template (np.ndarray): The template time series.
        fs (int): The sampling frequency in Hz.
        freqs (np.ndarray): Frequency array corresponding to the PSD.
        psd (np.ndarray): The Power Spectral Density of the noise.

    Returns:
        np.ndarray: The matched-filter output (signal-to-noise ratio time series).

    Examples:
        >>> fs = 4096
        >>> data = np.random.randn(fs)
        >>> template = np.sin(np.linspace(0, 10*np.pi, fs))
        >>> noise = np.random.randn(16 * fs)
        >>> freqs, psd = estimate_psd(noise, fs)
        >>> mf_output = matched_filter(data, template, fs, freqs, psd)
        >>> print(mf_output.shape)
        (4096,)
    """
    epsilon = 1e-48

    data_fft = np.fft.rfft(data)
    template_fft = np.fft.rfft(template)

    fft_freqs = np.fft.rfftfreq(len(data), 1.0 / fs)
    psd_interp = np.interp(fft_freqs, freqs, psd)

    data_whitened = data_fft / (np.sqrt(psd_interp) + epsilon)
    template_whitened = template_fft / (np.sqrt(psd_interp) + epsilon)

    cross_correlation = data_whitened * np.conj(template_whitened)

    # Inverse Fourier transform back to time domain, removed incorrect '* fs' scaling factor
    mf_output = np.fft.irfft(cross_correlation, len(data))

    df = fft_freqs[1] - fft_freqs[0]
    sigmasq = np.sum((np.abs(template_fft) ** 2 / (psd_interp + epsilon))) * df
    sigma = np.sqrt(np.abs(sigmasq))

    if sigma < epsilon:
        return np.zeros_like(mf_output)

    snr_timeseries = mf_output / sigma

    return snr_timeseries


def find_peak_snr(mf_output: np.ndarray) -> float:
    """
    Finds the peak Signal-to-Noise Ratio (SNR) from matched-filter output.
    Tag: [Metric calculation]

    Args:
        mf_output (np.ndarray): The time series output from the matched_filter function.

    Returns:
        float: The peak SNR value.

    Examples:
        >>> mf_out = np.array([1, 2, 5, 3, 1])
        >>> snr = find_peak_snr(mf_out)
        >>> print(snr)
        5.0
    """
    return np.max(np.abs(mf_output))

## Rationale

- The dataset prompt already contains the full code context and function names.
- Input rewriting can drop or distort code blocks, signatures, or unit-test
  evidence.
- Output rewriting can add prose or alter code formatting.
- The benchmark-specific role prompt should guide the agent to create local
  files, debug the code, and then return the completed functions directly.

Use `RH` or `RH--<llm-model-name>` in the OpenAI SDK request as usual.
